\section{Вклад конической точки в формуле Лефшеца в терминах преобразования Фурье-Лапласа}

\subsection{Общие определения}

Рассмотрим некомпактное многообразие \({ \hat{M} }\), составленное из компактного основания \({ M }\) с границей \({ \Omega }\) и присоединённого к нему бесконечного цилиндра \({ [0, +\infty) \times \Omega }\). Ему соответствует многообразие \({ \mathcal M }\) с конической точкой \({ \alpha }\). Рассмотрим на \({ \hat{M} }\) соболевские пространства \({ H^{s}(\hat{M}) \coloneqq H^{s, 0}(\mathcal M) }\). (См. \cite[p. 46, Definition 2.3]{nazaikinskii2005elliptic}.)

Пусть \[
    D : H^{s}(\hat{M}) \to H^{s - m}(\hat{M}) \]
--- дифференциальный оператор порядка \({ m }\), совпадающий на \({ [0, +\infty) \times \Omega }\) с оператором \({ D_{\infty} \coloneqq D_{\infty}(-i \partial_{t}, \omega, -i \partial_{\omega}) }\), \[
    D_{\infty} : H^{s}(\mathbb R_{t} \times \Omega) \to H^{s-m}(\mathbb R_{t} \times \Omega)\,.
\]

При сопряжении \({ D_{\infty} }\) с преобразованием Фурье мы получаем, фактически, конормальный символ оператора \({ D }\): \[
    \sigma_{c}(D)(p) = \mathcal F D_{\infty} \mathcal F^{-1} = D_{\infty}(p, \omega, -i \partial_{\omega})\,, \quad p \in \mathbb C\,.
\] Нам удобно вместо преобразования Фурье рассмотреть преобразование Фурье-Лапласа (см. \cite{taubes1987gauge} и \cite{mrowka2016index}): \[
    \begin{gathered}
        \mathcal F_{z} : L^2(\mathbb R_{t} \times \Omega) \to L^2(\mathbb S^{1}_{t} \times \mathbb S^{1}_{z} \times \Omega)\,, \\
        \left( \mathcal F_{z}u \right)(t, z, \omega) \coloneqq z^{\frac{t}{2\pi}} \sum_{n} z^{n} u(t + 2\pi n, \omega)\,.
    \end{gathered}
\] Обратное преобразование тогда задаётся формулой \[
    u(t, \omega) = \frac{1}{2\pi i} \oint_{\lvert z \rvert = 1} z^{-\frac{t}{2\pi}} (\mathcal F_{z}u)(t, z, \omega)\: \frac{dz}{z}\,.
\] При этом преобразовании оператор \({ D_{\infty} }\) переходит в семейство операторов \[
    \begin{gathered}
        \widetilde D_{\infty}(\theta) \coloneqq \mathcal F_{z} D_{\infty} \mathcal F_{z}^{-1} = D_{\infty}(-i \partial_{t} - \frac{\theta}{2\pi}, \omega, -i \partial_{\omega})\,, \\
        \widetilde D_{\infty}(\theta) : H^{s}(\mathbb S^{1}_{t} \times \Omega) \to H^{s-m}(\mathbb S^{1}_{t} \times \Omega)\,,
    \end{gathered}
\] где для удобства \({ z \coloneqq e^{i\theta} }\), \({ \theta \in [0, 2\pi] }\), а \({ H^{s}(\mathbb S^{1}_{t} \times \Omega) }\) суть расслоения над \({ \mathbb S^{1}_{\theta} }\) с функцией склейки \({ e^{it} }\).

\begin{definition}
    Оператор \({ D }\) называется эллиптическим, если 
    \begin{enumerate}
        \item \({ \sigma(D) }\) обратим на \({ T^{*}\hat{M} }\) вне нулевого сечения;
        \item \({ \widetilde D_{\infty}(\theta) }\) обратим для любого \({ \theta \in [0, 2\pi] }\).
    \end{enumerate}
\end{definition}

Из \cite[Теорема 2]{рабинович1982алгебре} следует

\begin{theorem}
    Если \({ D }\) --- эллиптический, то он фредгольмов.
\end{theorem}
    
В дальнейшем нам так же понадобится понятие следа оператора в смысле сужения его на некоторое подмногообразие (см. \cite{новиков1966следы}.) Пусть \[
    \begin{gathered}
        i : \Omega \hookrightarrow \mathbb R \times \Omega\,, \quad i(\omega) = (0, \omega)\,, \\
        A : C^{\infty}(\mathbb R \times \Omega) \to C^{\infty}(\mathbb R \times \Omega)\,.
    \end{gathered}
\]

\begin{definition}
    Следом оператора \({ A }\) на \({ \Omega }\) называется композиция \[
        \tau(A) : C^{\infty}(\Omega) \overset{i_*}{\longrightarrow} \mathcal D'(\mathbb R \times \Omega) \overset{A}{\longrightarrow} \mathcal D'(\mathbb R \times \Omega) \overset{i^{*}}{\longrightarrow} C^{\infty}(\Omega)\,,
    \] где \[
    \begin{gathered}
        (i_*u)(t, \omega) = u(\omega) \delta(t)\,, \\
        (i^{*}u)(\omega) = u(0, \omega)\,.
    \end{gathered}
    \]
\end{definition}

\subsection{Вклад конической точки \({ \alpha }\) в формуле Лефшеца\\\small(в терминах оператора сужения)}

Рассмотрим комплекс из одного эллиптического оператора 
\begin{equation}
    \label{eq:the_complex}
    0 \to H^{s}(\hat{M}) \overset{D}{\longrightarrow} H^{s-m}(\hat{M}) \to 0\,.
\end{equation} 
Пусть \({ T_{j} }\), \({ j = 1, 2 }\), --- геометрический эндоморфизм комплекса (\ref{eq:the_complex}), заданный диффеоморфизмом \({ \hat{M} \to \hat{M} }\), совпадающим на \({ [0, +\infty) \times \Omega }\) с отображением \({ g : \mathbb R_{t} \times \Omega \to \mathbb R_{t} \times \Omega }\) сдвига по переменной \({ t }\) на некоторое число \({ \lambda \neq 0 }\).

\begin{proposition}
    \label{prop:conical_contribution}
    Вклад неподвижной точки \({ \alpha }\) в формуле Лефшеца выражается как \[
        \mathcal L_{\operatorname{sing}} = i \operatorname{Trace} \int_{0}^{2\pi} \tau\left( \widetilde T_{1}(\theta) \widetilde D_{\infty} ^{-1}(\theta) \frac{\partial \widetilde D_{\infty}}{\partial \theta} (\theta) \right)\: d\theta\,,
    \] где \({ \widetilde T_{1}(\theta) = e^{-i \frac{\lambda \theta}{2 \pi}} g^{*} }\).
\end{proposition}

Для доказательства обозначим за \({ B = B(-i \partial_{t}, \omega, -i \partial_{\omega}) }\) псевдодифференциальный оператор с конормальным символом \[
    \sigma_{c}(D)^{-1}(p) \frac{\partial \sigma_{c}(D)}{\partial p}(p)\,.
    \] Для краткости будем опускать зависимость от \({ \omega }\) и писать, например, \({ B(-i \partial_{t}) }\) вместо \({ B(-i \partial_{t}, \omega, -i \partial_{\omega}) }\). Тогда, согласно \cite[Theorem 11.3]{nazaikinskii2005elliptic} (см. так же \cite{nazaikinskii1999atiyah}), 
\begin{equation}
    \label{eq:original_contribution}
    \mathcal L_{\operatorname{sing}} = \frac{1}{2 \pi i} \operatorname{Trace} \int\limits_{\mathclap{\mathbb R}} \sigma_{c}(T_1)(p) \, B_{\infty}(p)\: dp\,,
\end{equation}
где интеграл понимается в смысле его регуляризации \[
    \left( \frac{-1}{i \lambda} \right)^{l} \int_{-\infty}^{+\infty} \sigma_{c}(T_1)(p) \, B_{\infty}^{(l)}(p)\: dp
\] для достаточно большого \({ l }\).

\begin{lemma}
    Имеем, что 
    \begin{multline*}
        \left( \frac{-1}{i \lambda} \right)^{l} \int_{-\infty}^{+\infty} \sigma_{c}(T_1)(p) \,  B_{\infty}^{(l)}(p)\: dp \\ 
        = \left( \frac{-1}{i \lambda} \right)^{l} \int_{0}^{2\pi} \tau\left( \, e^{-i \frac{\lambda\theta}{2\pi}} g^{*} \, B_{\infty}^{(l)}\left( -i \partial_{t} - \frac{\theta}{2\pi} \right) \right) d\theta\,,
    \end{multline*}
    где след \({ \tau }\) отвечает вложению \({ \Omega \hookrightarrow \mathbb S^{1}_{t} \times \Omega }\).
\end{lemma}

\begin{proof}
    Подставляя разложение \({ \delta(t) = \frac{1}{2\pi} \sum_{k} e^{ikt} }\) в определение следа, получаем 
    \begin{multline*}
        \tau\left( \, e^{-i \frac{\lambda\theta}{2\pi}} g^{*} \, B_{\infty}^{(l)}\left( -i \partial_{t} - \frac{\theta}{2\pi} \right) \right) u(\omega) 
        \\
        \begin{aligned}
            &= \left[ e^{-i \frac{\lambda \theta}{2 \pi}} g^{*} \, B_{\infty}^{(l)}\left( -i \partial_{t} - \frac{\theta}{2\pi} \right) u(\omega) \delta(t) \right]_{t=0} \\
            &= \left[ \frac{1}{2\pi} e^{-i \frac{\lambda \theta}{2 \pi}} \sum_{k} B_{\infty}^{(l)}\left( k - \frac{\theta}{2\pi} \right) u(\omega) e^{ik(t + \lambda)} \right]_{t = 0} \\
            &= \left[ \frac{1}{2\pi} \sum_{k} e^{i (k \lambda - \frac{\lambda \theta}{2 \pi})} B_{\infty}^{(l)} \left( k - \frac{\theta}{2\pi} \right) \right] u(\omega)\,,
        \end{aligned}
    \end{multline*}
    то есть \[
        \tau\left( \, e^{-i \frac{\lambda\theta}{2\pi}} g^{*} \, B_{\infty}^{(l)}\left( -i \partial_{t} - \frac{\theta}{2\pi} \right) \right)
        = \frac{1}{2\pi} \sum_{k} e^{i \lambda (k - \frac{\theta}{2 \pi})} B_{\infty}^{(l)} \left( k - \frac{\theta}{2\pi} \right)\,.
    \]
    Отсюда
    \begin{multline*}
        \int_{0}^{2\pi} \tau\left( \, e^{-i \frac{\lambda\theta}{2\pi}} g^{*} \, B_{\infty}^{(l)}\left( -i \partial_{t} - \frac{\theta}{2\pi} \right) \right) d\theta \\
        = \frac{1}{2 \pi} \int_{0}^{2\pi} \sum_{k} e^{i \lambda (k - \frac{\theta}{2 \pi})} B_{\infty}^{(l)} \left( k - \frac{\theta}{2\pi} \right)\: d\theta 
        = \int_{-\infty}^{+\infty} e^{i \lambda p} B_{\infty}^{(l)} (p)\: dp\,.
    \end{multline*}
    Остаётся заметить, что \({ \sigma_{c}(T_{1}) = e^{i \lambda p} }\).
\end{proof}

Таким образом, если положить \({ \widetilde T_1(\theta) = e^{-i \frac{\lambda \theta}{2 \pi}} g^{*} }\), то вклад в формуле Лефшеца принимает вид 
\begin{equation}
    \label{eq:new_requilarized_contribution}
    \mathcal L_{\operatorname{sing}} = \frac{1}{2\pi i} \operatorname{Trace} \left( \frac{2 \pi}{i \lambda} \right)^{l} \int_{0}^{2\pi} \tau \left( \widetilde T_1(\theta) \widetilde B_{\infty}^{(l)}(\theta) \right)\: d\theta\,,
\end{equation}
поскольку \[
    \widetilde B_{\infty}^{(l)}(\theta) 
    = \frac{\partial^{l}}{\partial \theta^{l}} \left( B_{\infty}\left( -i \partial_{t} - \frac{\theta}{2\pi} \right) \right)
    = \left( \frac{-1}{2\pi} \right)^{l} B_{\infty}^{(l)}\left( -i \partial_{t} - \frac{\theta}{2\pi} \right)\,.
\] Наконец, интегрируя по частям в выражении (\ref{eq:new_requilarized_contribution}), получаем \[
    \mathcal L_{\operatorname{sing}} = \frac{1}{2\pi i} \operatorname{Trace} \int_{0}^{2\pi} \tau \left( \widetilde T_{1}(\theta) \, \widetilde B_{\infty}(\theta) \right)\: d\theta\,,
\] что есть в точности равенство из Утверждения~(\ref{prop:conical_contribution}), поскольку \[
    \widetilde B_{\infty}(\theta) = -2\pi \widetilde D_{\infty} ^{-1}(\theta) \frac{\partial \widetilde D_{\infty}}{\partial \theta} (\theta)\,.
\] 

\begin{remark}
    Вклад можно так же переписать в терминах цилиндра \({ \mathbb R_{t} \times \Omega }\): \[
        \mathcal L_{\operatorname{sing}} = \frac{1}{2\pi i} \operatorname{Trace} \tau \left( T_{1\infty} D^{-1}_{\infty} [D_{\infty}, t] \right)\,,
    \] где след \({ \tau }\) отвечает вложению \({ \Omega \hookrightarrow \mathbb R_{t} \times \Omega\,, \ \omega \mapsto (0, \omega) }\). Это следует из формулы (\ref{eq:original_contribution}) и соотношения \[
        \tau(D_{\infty}) = \int_{\mathbb R} \sigma_{c}(D)\: dp\,,
    \] непосредственно вытекающего из определений\footnote{По определению \({ \tau(D_{\infty}) u(\omega) = \left[ D_{\infty} \delta(t) u(\omega) \right]_{t = 0}^{}  }\).
       \[
           \begin{aligned}
               (D_{\infty} \delta(t)u(\omega), \varphi(t, \omega)) 
               &= \left(\mathcal F_{p \to t}^{-1} \sigma_{c}(D) \mathcal F_{t \to p} \delta(t) u(\omega), \varphi(t, \omega)\right) \\
               &= \left(\delta(t)u(\omega), \mathcal F_{p \to t} \sigma_{c}(D) \mathcal F_{t \to p}^{-1} \varphi(t, \omega)\right) \\
               &= u(\omega) \int_{\mathbb R} \varphi(\tau, \omega) \int_{\mathbb R} e^{i p \tau} \sigma_{c}(D)\: dp\: d\tau \\
               &= \left(\mathcal F_{p \to \tau}^{-1} \sigma_{c}(D) u(\omega), \varphi(\tau, \omega)\right)\,.
           \end{aligned}
        \] Отсюда \({ \tau(D_{\infty}) = \mathcal F_{p \to t}^{-1} \sigma_{c}(D) \Big |_{t = 0}^{} = \int_{\mathbb R} \sigma_{c}(D)\: dp }\).
    }.
\end{remark}

\begin{remark}
    Можно так же рассматривать и более общие диффеоморфизмы вида \[
        \begin{gathered}
            g : (t, \omega) \mapsto (t + \lambda(\omega), g_{\alpha}(\omega))\,,  \\
            \lambda : \Omega \to \mathbb R \setminus \left\{ 0 \right\}\,, \quad 
            g_{\alpha} : \Omega \to \Omega\,.
        \end{gathered}
    \] Утверждение~(\ref{prop:conical_contribution}) в таком случае остаётся в точности тем же, просто \({ \lambda }\) в выражении для \({ \widetilde T_1(\theta) }\) будет зависеть от \({ \omega }\).
\end{remark}

\subsection{Вклад бесконечности для многообразий с периодическими концами}

Пусть \({ \mathcal M }\) --- риманово многообразие размерности \({ n }\) с периодическим концом, отвечающим бесконечному циклическому накрытию \({ M }\) компактного многообразия \({ M_{\infty} }\), ассоциированному с классом когомологий \({ \gamma \in H^{1}(M_{\infty}; \mathbb Z) }\), двойственным (по Пуанкаре) ориентированному гладкому подмногообразию \({ Y }\) коразмерности \({ 1 }\). Тогда \({ \mathcal M }\) состоит из компактного основания \({ M_{0} }\) с границей \({ Y }\) и хвоста \({ M^{+} }\), представляющего из себя <<половину>> \({ M }\), отрезанную по \({ Y }\) (см. \cite{mrowka2015index}.) 

При этом \({ M }\) можно представить как склейку по отображениям \({ Y^{-} \to Y^{+} }\) счётного числа фундаментальных областей \({ W_{k} }\), изометрически изоморфных \({ M_{\infty} \setminus Y }\) и имеющих два берега разреза \({ Y^{\pm} }\).
Бесконечная циклическая группа \({ \Pi \coloneqq \langle t \rangle }\) свободно действует на \({ M }\) естественным сдвигом, \({ t W_{k} = W_{k+1} }\). Для \({ k \in \mathbb Z_{+} }\) обозначим \({ M^{+}_{k} \coloneqq t^{k} M^{+} \subseteq M^{+} \subseteq \mathcal M }\).

Легко построить гладкую функцию \({ f : M \to \mathbb R }\) такую, что \begin{equation}
    \label{eq:quasi-periodicity_of_f}
    f(tx) = f(x) + 1\,.
\end{equation} Достаточно положить \({ f }\) гладкой ступенькой в односторонней трубчатой окрестности \({ Y^{+} }\) в \({ W_{0} }\), продолжить \({ f }\) нулём в \({ W_{0} }\) и определить соотношением (\ref{eq:quasi-periodicity_of_f}) на всём \({ M }\). При этом \({ df }\) корректно определено на \({ M_{\infty} }\).

Пусть \({ g : \mathcal M \to \mathcal M }\) есть диффеоморфизм, для достаточно большого \({ N }\) совпадающий на \({ M^{+}_{N} }\) с некоторым \({ g^{\infty} : M \to M }\), коммутирующим с действием \({ \mathbb Z }\) и таким, что \({ g(M^{+}_{N}) \subseteq M^{+} }\). (Возможно, для большей общности вместо периодичности можно требовать что-то вроде существования равномерного предела \({ \lim_{k \to \infty} t^{-k} g t^{k} (x) }\).) Тогда \({ g^{\infty} }\) индуцирует отображение \({ g_{\infty} : M_{\infty} \to M_{\infty} }\).

\begin{remark}
    \label{rm:fixed_points_are_projections}
    Не умаляя общности считаем, что все неподвижные точки \({ g_{\infty} }\) суть проекции неподвижных точек \({ g^{\infty} }\). В противном случае можно вместо \({ M_{\infty} = M / \langle t \rangle }\) взять \({ M / \langle t^{k + 1} \rangle }\) для наибольшего по модулю \({ k \in \mathbb Z }\) такого, что \({ g(t^{k} x) = x }\) для некоторого \({ x \neq g(x) }\).
\end{remark}

Ожидается, что справедлива следующая

\begin{theorem}
    \label{thm:preiodic_singular_contribution}
    Пусть \({ g }\) имеет только невырожденные неподвижные точки, а \({ g^{\infty} }\) не имеет неподвижных точек. Тогда число Лефшеца \({ 
        \mathcal L = \mathcal L_{\operatorname{int}} + \mathcal L_{\operatorname{sing}} 
    }\) где \({ \mathcal L_{\operatorname{int}} }\) есть сумма классических вкладов неподвижных точек \({ g }\), лежащих внутри \({ \mathcal M \setminus M^{+}_{N} }\), а 
    \begin{equation}
        \label{eq:L_sing}
        \mathcal L_{\operatorname{sing}} = i \operatorname{Trace}_{M_{\infty}} \int_{0}^{2 \pi} T_{1}(\theta) D^{^{-1}}_{\infty}(\theta) \frac{\partial D_{\infty}}{\partial \theta}(\theta) d\theta\,.
    \end{equation}
\end{theorem}

Теперь рассмотрим случай, когда \({ g^{\infty} }\) имеет невырожденные неподвижные точки. Пусть \({ \left\{ x_{j} \right\} }\) --- неподвижные точки \({ g_{\infty} }\) (тоже невырожденные в силу Замечания~\ref{rm:fixed_points_are_projections}) и пусть все \({ x_{j} }\) лежат вне \({ Y }\). В этом случае на \({ \mathcal M }\) имеется бесконечное количество неподвижных точек, но вклад (\ref{eq:L_sing}) всё ещё корректно определён. Однако, возникает проблема в неоднозначности выбора разреза \({ Y }\) в \({ \mathcal M }\), от которого может зависеть набор неподвижных точек, которые мы относим к ``внутренним'' и учитываем в \({ \mathcal L_{\operatorname{int}} }\). Эту неоднозначность можно учесть путём добавления дополнительного слагаемого в формулу Лефшеца.

Именно, пусть \({ \omega \in (\Omega^1(M_{\infty}))' }\) --- поток де Рама степени \({ n-1 }\) на \({ M_{\infty} }\) (в смысле \cite[Chapter III]{de2012differentiable}), для которого \[
    d \omega = \sum_{x_{j}} \mathcal L_{x_{j}} \delta_{x_{j}} \in (C^{\infty}(M_{\infty}))'\,,
\] где \({ \mathcal L_{x_{j}} }\) есть классический вклад неподвижной точки \({ x_{j} }\), а \({ \delta_{x_{j}} }\) --- \({ \delta }\)-функция, сосредоточенная в точке \({ x_{j} }\) и воспринимаемая как поток старшей степени на \({ M_{\infty} }\). Такой поток \({ \omega }\) существует, поскольку по условию эллиптичности оператор \({ D_{\infty} }\) обратим и поэтому его число Лефшеца, равное \({ \sum_{x_{j}} \mathcal L_{x_{j}} }\), равно нулю.

Тогда, по аналогии с \cite[Theorem A]{mrowka2016index}, гипотетическая формула для числа Лефшеца принимает вид \({ \mathcal L = \mathcal L_{\operatorname{int}} + \mathcal L' + \mathcal L_{\operatorname{sing}} }\), где \[
    \mathcal L' = -\int_{Y} \omega + \int_{M_{\infty}} df \land \omega\,.
\]

\begin{remark}
    По определению внешнего произведения для потоков и по определению интеграла от потока имеем \[
        \int_{M_{\infty}} df \land \omega = (-1)^{\deg \omega} \omega \land df [1] = (-1)^{n-1} \omega[df \land 1] = (-1)^{n-1} \omega[df]\,,
    \] где \({ \omega[\varphi] }\) обозначает значение \({ \omega }\) на форме \({ \varphi \in \Omega^{1}(M_{\infty}) }\). При этом \({ \omega }\) по построению гладка вне неподвижных точек \({ x_{j} }\), поэтому её сужение \({ \omega |_{Y} }\) корректно определено и интеграл \({ \int_{Y} \omega }\) имеет смысл.
\end{remark}

\begin{remark}
    \({ \mathcal L' }\) не зависит от выбора \({ \omega }\), поскольку для \({ d\omega = d(\omega + \omega') }\) в силу формулы интегрирования по частям и определения функции \({ f }\) имеем \[
        \int_{M_{\infty}} df \land \omega' - \int_{Y} \omega' = -\int_{W_0} f \land \underbrace{d\omega'}_{0} + \int_{Y} (\underbrace{ft - f}_{1}) \omega' - \int_{Y} \omega' = 0\,.
    \]
\end{remark}

\begin{remark}
    Из формулы Стокса (если она справедлива для потоков) следует, что если при построении \({ M }\) вместо \({ Y }\) взять какой-то другой разрез \({ \widetilde Y }\), то к \({ \mathcal L' }\) прибавляются с соответствующими знаками вклады неподвижных точек, заключённых между \({ Y }\) и \({ \widetilde Y }\). Это компенсирует ``лишние'' вклады, добавляющиеся при этом в \({ \mathcal L_{\operatorname{int}} }\).
\end{remark}

\subsection{Вклад особых точек в случае комплекса де~Рама}

Пусть, снова, \({ \mathcal M }\) --- риманово многообразие размерности \({ n }\) с периодическим концом, отвечающим бесконечному циклическому накрытию \({ M }\) компактного многообразия \({ M_{\infty} }\). Тогда \({ \mathcal M }\) состоит из компактного основания \({ M_0 }\) с границей \({ Y }\) и хвоста \({ M^{+} }\). Мы будем так же предполагать, что \({ H_{*}(\mathcal M; \mathbb C) }\) конечномерны.

По аналогии с \cite{mrowka2015index}, для $\delta \in \mathbb R$ обозначим через $\Omega_{\delta}^q(\mathcal M)$ пополнение пространства гладких финитных $q$-форм на $\mathcal M$ по норме, совпадающей на периодическом конце с
\[
    \|\alpha\|_{L^2_{\delta}}^2 \coloneqq \int_{M^{+}} e^{\delta f(x)} |\alpha(x)|^2\, dx\,.
\]
Соответствующий \emph{взвешенный комплекс де~Рама} имеет вид
\begin{equation}
    \label{eq:weighted_de_rham_complex}
    0 \to \Omega_{\delta}^0(\mathcal M)
    \overset{d}{\longrightarrow} \Omega_{\delta}^1(\mathcal M)
    \overset{d}{\longrightarrow} \cdots
    \overset{d}{\longrightarrow} \Omega_{\delta}^n(\mathcal M)
    \to 0\,.
\end{equation}
Его гомологии будут обозначаться \({ H^{*}_{\delta}(\mathcal M) }\).

\begin{lemma}
    \label{lem:twisted_de_rham_differential}
    Преобразование Фурье--Лапласа на периодическом конце переводит оператор $d$ в
    \begin{equation}
        \label{eq:twisted_differential}
        d_z \coloneqq \mathcal F_{z} d \mathcal F_{z}^{-1} = d - (\ln z)\, df \wedge \cdot\,.
    \end{equation}
\end{lemma}

Получаем ``скрученный'' комплекс \[
    \cdots \to \Omega^{k}(M_{\infty}) \xrightarrow{d - \ln z\: df \land} \Omega^{k+1}(M_{\infty}) \to \cdots\,,
\] чьи когомологии (с коэффициентами в комплексном одномерном расслоении с плоской связностью \({ -\ln z\: df \land }\)) будут обозначаться \({ H_{z}^{*}(M_{\infty}) }\).

\begin{definition}
    \label{def:exceptional_set}
    \emph{Исключительным множеством} комплекса де~Рама на периодическом конце назовём множество
    \[
        \Sigma \coloneqq \left\{ z \in \mathbb C^* : H^{*}_z(M_{\infty}) \neq 0 \right\}\,,
    \]
    где $H^{*}_z(M_{\infty})$ обозначает когомологии комплекса $(\Omega^{*}(M_{\infty}), d_z)$.
\end{definition}

\begin{proposition}
    \label{prop:fredholm_criterion_derham}
    Комплекс~\eqref{eq:weighted_de_rham_complex} является фредгольмовым тогда и только тогда, когда окружность $C_{\delta}$ не пересекает множество $\Sigma$. В частности, исключительные веса образуют конечное множество значений $\delta = \ln |z_0|$, где $z_0 \in \Sigma$.
\end{proposition}

\begin{proof}
    Это утверждение есть переформулировка~\cite[Propositions 2.1 \& 2.2]{mrowka2015index}.
\end{proof}

\begin{proposition}
    \label{prop:explicit_description_of_the_critical_set}
    Множество \({ \Sigma }\) есть в точности множество собственных значений оператора \({ t_{*} : H_{*}(M; \mathbb C) \to H_{*}(M; \mathbb C) }\).
\end{proposition}

\begin{proof}
    Следует из~\cite[Proof of Theorem 1.1]{mrowka2015index}.
\end{proof}

Из \cite[Propositions 2.2 \& 2.4]{mrowka2015index} следует, что для того, чтобы комплекс (\ref{eq:weighted_de_rham_complex}) был фредгольмовым хотя бы при одном значении \({ \delta }\), необходимо и достаточно, чтобы гомологии \({ H_*(M; \mathbb C) }\) были конечномерными. Оттуда же \cite[Propoistion 2.2]{mrowka2015index} следует, что если \({ \Sigma }\) непусто, то дискретно.

\begin{definition}
    Если комплекс~\eqref{eq:weighted_de_rham_complex} фредгольмов, то его \emph{взвешенным числом Лефшеца} относительно геометрического эндоморфизма, индуцированного отображением $g$, называется величина
    \[
        \mathcal L^{\operatorname{dR}}_{\delta}(g)
        \coloneqq
        \sum_{q=0}^{n} (-1)^q
        \operatorname{Tr}\bigl(
            g^* \colon H^q_{\delta}(\mathcal M) \to H^q_{\delta}(\mathcal M)
        \bigr).
    \]
\end{definition}

\subsubsection{Мотивация для формулы}

В простейшем случае, когда \({ g }\) индуцирует тождественные отображения в когомологиях \({ H^{*}_{\delta}(\mathcal M) }\), его число Лефшеца равно в точности индексу \({ \operatorname{ind}_{\delta}(\mathcal M) }\) комплекса~(\ref{eq:weighted_de_rham_complex}).

\begin{theorem}
    \({ \operatorname{ind}_{\delta}(\mathcal M) }\) есть кусочно-постоянная функция от \({ \delta }\), скачки которой имеют место при \({ \delta = \ln \lvert \lambda \rvert }\), где \({ \lambda \in \Sigma }\). Для каждого \({ k = 0, \ldots, n-1 }\) каждое такое \({ \lambda }\) даёт вклад в скачёк в размере \({ (-1)^{k+1} }\), умноженного на его кратность как корня характеристического многочлена оператора \[
        t_{*} : H_{k}(M; \mathbb C) \to H_{k}(M; \mathbb C)\,.
    \]
\end{theorem}

\begin{proof}
    \cite[Theorem 1.2]{mrowka2015index}.
\end{proof}

Можно заметить, что сумма вкладов в предыдущей теореме представляется как альтерированная сумма следов отображения \({ g }\) в корневых подпространствах оператора \({ t_{*} }\).

\subsubsection{Гипотетическая формула}

\begin{definition}
    Корневое подпространство пространство оператора \[
        t_* : H_{*}(M; \mathbb C) \to H_{*}(M; \mathbb C)\,,
    \] отвечающее \({ z \in \mathbb C }\), будем называть спектральным блоком и обозначать \({ \mathcal H_{*}^{z}M }\).
\end{definition}

Поскольку \({ g_{\infty} }\) коммутирует с оператором сдвига \({ t }\), то индуцированное отображение \({ g_{\infty, *} }\) сохраняет корневые подпространства оператора \({ t_{*} }\). Поэтому, имеют место корректно определённые индуцированные отображения \[
    g_{\infty, *} : \mathcal H_{*}^{z}M \to \mathcal H_{*}^{z}M\,.
\]

\begin{definition}
    \label{def:spectral_lefschetz_contribution}
    Для $z \in \Sigma$ определим \emph{спектральный вклад на бесконечности} отображения \({ g_{\infty} }\) формулой
    \[
        \Lambda_z(g_{\infty}) \coloneqq \sum_{k=0}^{n} (-1)^{k+1} \operatorname{Tr}\bigl( g_{\infty,*} \colon \mathcal H^k_z(M) \to \mathcal H^k_z(M) \bigr)\,.
    \]
\end{definition}

Гипотетически, имеет место следующая

\begin{theorem}
    \label{thm:periodic_de_rham_formula}
    Предположим, что отображение $g$ имеет только невырожденные неподвижные точки, а \({ g_{\infty} }\) не имеет неподвижных точек. Пусть также $\delta_1, \delta_2 \in \mathbb R$ выбраны так, что соответствующие комплексы~\eqref{eq:weighted_de_rham_complex} фредгольмовы. Тогда
    \begin{equation}
        \label{eq:weighted_periodic_lefschetz_jump_formula}
        \mathcal L^{\operatorname{dR}}_{\operatorname{sing},\delta_2} - \mathcal L^{\operatorname{dR}}_{\operatorname{sing},\delta_1}
        =
        \sum_{\substack{z_0 \in \Sigma \\ e^{\delta_1} < |z_0| < e^{\delta_2}}}
        \Lambda_{z_0}(g_{\infty})\,.
    \end{equation}
\end{theorem}

\begin{proof}[Набросок доказательства]
    Для применения Теоремы~\ref{thm:preiodic_singular_contribution} к комплексу~\eqref{eq:weighted_de_rham_complex}
    будем рассматривать его \(2\)-периодизацию \[
        \Omega^{\operatorname{ev}}_{\delta}(M) \overset{D}{\longrightarrow} \Omega^{\operatorname{odd}}_{\delta}(M) \overset{D}{\longrightarrow} \Omega^{\operatorname{ev}}_{\delta}(M)\,,
        \qquad
        D = d + d^*,
    \] где \(d^*\) есть формально сопряжённый к \(d\).
    После преобразования Фурье--Лапласа на периодическом
    конце получаем семейство операторов Ходжа--де~Рама
    \[
        D_z := d_z + d_z^*\,.
    \]
    При этом \(D_z\) обратим тогда и только тогда, когда \(H_z^*(M_\infty)=0\), что эквивалентно \({ z \not \in \Sigma }\).

    Формулу~\eqref{eq:L_sing} для вклада бесконечности после замены \({ z = e^{\delta + i \theta} }\), предположительно, можно переписать в виде
    \begin{equation}
        \label{eq:de_rham_sing_hodge}
        \mathcal L^{\operatorname{dR}}_{\operatorname{sing},\delta} =
        \frac{1}{2\pi i} \int_{C_\delta} \operatorname{Trace} \left(
            g_\infty^* D_z^{-1}\frac{\partial D_z}{\partial z}
        \right)\,dz\,.
    \end{equation}

    Подынтегральная функция мероморфна на \(\mathbb C^*\), и её полюса лежат ровно в точках множества \(\Sigma\). Следовательно, по теореме о вычетах вычисление скачка \({ \mathcal L^{\operatorname{dR}}_{\operatorname{sing},\delta_2} - \mathcal L^{\operatorname{dR}}_{\operatorname{sing},\delta_1} }\) сводится к вычислению вычетов подынтегральной функции в точках множества \({ \Sigma }\), заключённых в соответствующем кольце.
\end{proof}

\begin{remark}
    Даже если приведённая теорема верна, для полного описания значений \({ \mathcal L^{\operatorname{dR}}_{\operatorname{sing}, \delta} }\) всё ещё не хватает известного значения \({ \mathcal L^{\operatorname{dR}}_{\operatorname{sing}, \delta} }\) по крайней мере при одном \({ \delta }\), либо же при \({ \delta \to \pm \infty }\).
\end{remark}
